#include <stdio.h>

void bubble_sort(int array[], int n) {
    for (int i=0; i<n-1; i++) {
        for (int j=0; j<n-i-1; j++) {
            if (array[j] > array[j+1]) {
                int temp = array[j];
                array[j] = array[j+1];
                array[j+1] = temp;
            }
        }
    }
}

int main(){
	int array[10];
	int x, i;
	
	for(int i=0; i <10; i++){
		printf("請輸入第%d個數:", i+1);

	printf("顯示這10個數:\n");
	for(int i=0; i <10; i++){
		printf("%d, ", array[i]);
	}
	printf("\n");

	bubble_sort(array, 10);

    printf("排序後 = ");

}
