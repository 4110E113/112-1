#include <stdio.h>

void bubble_sort(int array[], int n) {
    for (int i=0; i<n-1; i++) {
        for (int j=0; j<n-i-1; j++) {
            if (array[j] > array[j+1]) {
                int temp = array[j];
                array[j] = array[j+1];
                array[j+1] = temp;
            }
        }
    }
}

int main(){
	int array[10];

#include "stdio.h"
#include "stdlib.h"
#include "time.h"
#include "BubbleSort.h"

int main(){
	int data[20];
	for(int i=0;i<20 ;i++){
		int m=0, n=1;
		while(n==1){
			n=0;
			srand((unsigned)time(NULL));
			m = (rand()*(rand()%10))%100;
			// 以時間序列當亂數種子 取0-99的亂數 
			for(int j=0;j<i;j++){
				if(m==data[j]) n=1;
			}	
		}
		data[i]=m;
	}
	printf("有20筆資料值:\n"); 
	for(int i=0;i<20 ;i++){
		printf("%d, ", data[i]);
	}
	printf("\n\n");
	BubbleSorting(data, 0, 19);
	printf("使用氣泡排序法(Bubble Sort)排列後:\n");
	for(int i=0;i<20 ;i++){
		printf("%d, ", data[i]);
	}
	printf("\n");
	return 0;	
}

int BubbleSorting(int data[20], int f, int l){
	for(int i=l; i>f; i--){
		BubbleMove(data, f, i);
	}	
}

int BubbleMove(int data[20], int f, int l){
	for(int i=f; i<l; i++){
		if (data[i+1]<data[i]){
			int k = data[i+1];
			data[i+1]=data[i];
			data[i]=k;
		}
	}
	return 0;
}
