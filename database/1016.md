### group by
```sql
SELECT	ksu_std_name as 學生姓名,
		AVG(ksu_std_grade) as 平均成績, 
        COUNT(*) as 人數
FROM `ksu_std_table` 
GROUP BY ksu_std_name;
```
### having
```spl
SELECT	ksu_std_name as 學生姓名,
		AVG(ksu_std_grade) as 平均成績, 
        COUNT(*) as 人數
FROM `ksu_std_table` 
GROUP BY ksu_std_name
HAVING AVG(ksu_std_grade)>50 and 人數>1;
```
