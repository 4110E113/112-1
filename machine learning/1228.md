import cv2
import tensorflow_hub as hub
import numpy
import pandas as pd
import tensorflow as tf

detector = hub.load("d:/prog/efficientdet_lite2_detection_1")

labels = labels['OBJECT (2017 REL.)']

    pred_labels=classes.numpy().astype('int')[0]
